\subsubsection{Assignment Statements}
ASSIGN\_OPRTR = \{\cd{=, +, -=, *=, /=, \%=}\}\\\\
\noindent The syntax is the same with other common languages:
\begin{minted}{text}
<IDENT> <ASSIGN_OPRTR> <VALUE>
\end{minted}
\bd{Example:}
\begin{minted}{text}
pi = 3.14; radius = 45;
radius *= radius;    -- radius = radius * 45 = 2025
area = pi * radius;  -- area = 3.14 * 2025 = 6358.5
\end{minted}
